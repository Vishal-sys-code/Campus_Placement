
iomanip is a library that is used to manipulate the output of C++ program.

Q.)
#include <iostream>
#include <iomanip>
int main () 
{
  std::cout << std::showpos<<-100;
  return 0;
}

Output: -100


Q.)
#include <iostream>
#include <iomanip>
int main () {
  int x = 786;
  std::cout<<std::setfill('*')<<std::setw(6)<<x;
  return 0;
}

Output: ***786
